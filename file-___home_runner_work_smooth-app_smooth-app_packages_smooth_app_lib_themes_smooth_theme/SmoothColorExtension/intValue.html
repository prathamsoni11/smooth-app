int get intValue {
  final int a = (this.a * 255).round();
  final int r = (this.r * 255).round();
  final int g = (this.g * 255).round();
  final int b = (this.b * 255).round();

  // Combine the components into a single int using bit shifting
  return (a << 24) | (r << 16) | (g << 8) | b;
}
